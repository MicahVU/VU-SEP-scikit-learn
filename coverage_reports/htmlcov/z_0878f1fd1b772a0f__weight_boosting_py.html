1"""Weight Boosting. 

2 

3This module contains weight boosting estimators for both classification and 

4regression. 

5 

6The module structure is the following: 

7

9 for all the estimators in the module. Regression and classification 

10 only differ from each other in the loss function that is optimized.

13 (AdaBoost-SAMME) for classification problems.

318def _samme_proba(estimator, n_classes, X): 

319 """Calculate algorithm 4, step 2, equation c) of Zhu et al [1]. 

320 

321 References 

322 ---------- 

323 .. [1] J. Zhu, H. Zou, S. Rosset, T. Hastie, "Multi-class AdaBoost", 2009. 

324 

325 """ 

326 proba = estimator.predict_proba(X) 

327 

328 # Displace zero probabilities so the log is defined. 

329 # Also fix negative elements which may occur with 

330 # negative sample weights. 

331 np.clip(proba, np.finfo(proba.dtype).eps, None, out=proba) 

332 log_proba = np.log(proba) 

333 

334 return (n_classes - 1) * ( 

335 log_proba - (1.0 / n_classes) * log_proba.sum(axis=1)[:, np.newaxis] 

336 )

339class AdaBoostClassifier( 

340 _RoutingNotSupportedMixin, ClassifierMixin, BaseWeightBoosting 

341): 

342 """An AdaBoost classifier. 

343 

344 An AdaBoost [1]_ classifier is a meta-estimator that begins by fitting a 

345 classifier on the original dataset and then fits additional copies of the 

346 classifier on the same dataset but where the weights of incorrectly 

347 classified instances are adjusted such that subsequent classifiers focus 

348 more on difficult cases. 

349 

350 This class implements the algorithm based on [2]_. 

351 

352 Read more in the :ref:`User Guide <adaboost>`. 

353 

354 .. versionadded:: 0.14 

355 

356 Parameters 

357 ---------- 

358 estimator : object, default=None 

359 The base estimator from which the boosted ensemble is built. 

360 Support for sample weighting is required, as well as proper 

361 ``classes_`` and ``n_classes_`` attributes. If ``None``, then 

362 the base estimator is :class:`~sklearn.tree.DecisionTreeClassifier` 

363 initialized with `max_depth=1`. 

364 

365 .. versionadded:: 1.2 

366 `base_estimator` was renamed to `estimator`. 

367 

368 n_estimators : int, default=50 

369 The maximum number of estimators at which boosting is terminated. 

370 In case of perfect fit, the learning procedure is stopped early. 

371 Values must be in the range `[1, inf)`. 

372 

373 learning_rate : float, default=1.0 

374 Weight applied to each classifier at each boosting iteration. A higher 

375 learning rate increases the contribution of each classifier. There is 

376 a trade-off between the `learning_rate` and `n_estimators` parameters. 

377 Values must be in the range `(0.0, inf)`. 

378 

379 algorithm : {'SAMME', 'SAMME.R'}, default='SAMME.R' 

380 If 'SAMME.R' then use the SAMME.R real boosting algorithm. 

381 ``estimator`` must support calculation of class probabilities. 

382 If 'SAMME' then use the SAMME discrete boosting algorithm. 

383 The SAMME.R algorithm typically converges faster than SAMME, 

384 achieving a lower test error with fewer boosting iterations. 

385 

386 .. deprecated:: 1.4 

387 `"SAMME.R"` is deprecated and will be removed in version 1.6. 

388 '"SAMME"' will become the default. 

389 

390 random_state : int, RandomState instance or None, default=None 

391 Controls the random seed given at each `estimator` at each 

392 boosting iteration. 

393 Thus, it is only used when `estimator` exposes a `random_state`. 

394 Pass an int for reproducible output across multiple function calls. 

395 See :term:`Glossary <random_state>`. 

396 

397 Attributes 

398 ---------- 

399 estimator_ : estimator 

400 The base estimator from which the ensemble is grown. 

401 

402 .. versionadded:: 1.2 

403 `base_estimator_` was renamed to `estimator_`. 

404 

405 estimators_ : list of classifiers 

406 The collection of fitted sub-estimators. 

407 

408 classes_ : ndarray of shape (n_classes,) 

409 The classes labels. 

410 

411 n_classes_ : int 

412 The number of classes. 

413 

414 estimator_weights_ : ndarray of floats 

415 Weights for each estimator in the boosted ensemble. 

416 

417 estimator_errors_ : ndarray of floats 

418 Classification error for each estimator in the boosted 

419 ensemble. 

420 

421 feature_importances_ : ndarray of shape (n_features,) 

422 The impurity-based feature importances if supported by the 

423 ``estimator`` (when based on decision trees). 

424 

425 Warning: impurity-based feature importances can be misleading for 

426 high cardinality features (many unique values). See 

427 :func:`sklearn.inspection.permutation_importance` as an alternative. 

428 

429 n_features_in_ : int

431 

432 .. versionadded:: 0.24 

433

555 """Implement a single boost. 

556 

557 Perform a single boost according to the real multi-class SAMME.R 

558 algorithm or to the discrete SAMME algorithm and return the updated 

559 sample weights. 

560 

561 Parameters 

562 ---------- 

563 iboost : int 

564 The index of the current boost iteration. 

565 

566 X : {array-like, sparse matrix} of shape (n_samples, n_features) 

567 The training input samples. 

568 

569 y : array-like of shape (n_samples,) 

570 The target values (class labels). 

571 

572 sample_weight : array-like of shape (n_samples,) 

573 The current sample weights. 

574 

575 random_state : RandomState instance 

576 The RandomState instance used if the base estimator accepts a 

577 `random_state` attribute. 

578 

579 Returns 

580 ------- 

581 sample_weight : array-like of shape (n_samples,) or None 

582 The reweighted sample weights. 

583 If None then boosting has terminated early. 

584 

585 estimator_weight : float 

586 The weight for the current boost. 

587 If None then boosting has terminated early. 

588 

589 estimator_error : float 

590 The classification error for the current boost. 

591 If None then boosting has terminated early. 

592 """ 

593 if self.algorithm == "SAMME.R": 

594 return self._boost_real(iboost, X, y, sample_weight, random_state) 

595 

596 else: # elif self.algorithm == "SAMME": 

597 return self._boost_discrete(iboost, X, y, sample_weight, random_state)

599 # TODO(1.6): Remove function. The `_boost_real` function won't be used any 

600 # longer, because the SAMME.R algorithm will be deprecated in 1.6. 

601 def _boost_real(self, iboost, X, y, sample_weight, random_state): 

602 """Implement a single boost using the SAMME.R real algorithm.""" 

603 estimator = self._make_estimator(random_state=random_state) 

604 

605 estimator.fit(X, y, sample_weight=sample_weight) 

606 

607 y_predict_proba = estimator.predict_proba(X) 

608 

609 if iboost == 0: 

610 self.classes_ = getattr(estimator, "classes_", None) 

611 self.n_classes_ = len(self.classes_) 

612 

613 y_predict = self.classes_.take(np.argmax(y_predict_proba, axis=1), axis=0) 

614 

615 # Instances incorrectly classified 

616 incorrect = y_predict != y 

617 

618 # Error fraction 

619 estimator_error = np.mean(np.average(incorrect, weights=sample_weight, axis=0)) 

620 

621 # Stop if classification is perfect 

622 if estimator_error <= 0: 

623 return sample_weight, 1.0, 0.0 

624 

625 # Construct y coding as described in Zhu et al [2]: 

626 # 

627 # y_k = 1 if c == k else -1 / (K - 1) 

628 # 

629 # where K == n_classes_ and c, k in [0, K) are indices along the second 

630 # axis of the y coding with c being the index corresponding to the true 

631 # class label. 

632 n_classes = self.n_classes_ 

633 classes = self.classes_ 

634 y_codes = np.array([-1.0 / (n_classes - 1), 1.0]) 

635 y_coding = y_codes.take(classes == y[:, np.newaxis]) 

636 

637 # Displace zero probabilities so the log is defined. 

638 # Also fix negative elements which may occur with 

639 # negative sample weights. 

640 proba = y_predict_proba # alias for readability 

641 np.clip(proba, np.finfo(proba.dtype).eps, None, out=proba) 

642 

643 # Boost weight using multi-class AdaBoost SAMME.R alg 

644 estimator_weight = ( 

645 -1.0 

646 * self.learning_rate 

647 * ((n_classes - 1.0) / n_classes) 

648 * xlogy(y_coding, y_predict_proba).sum(axis=1) 

649 ) 

650 

651 # Only boost the weights if it will fit again 

652 if not iboost == self.n_estimators - 1: 

653 # Only boost positive weights 

654 sample_weight *= np.exp( 

655 estimator_weight * ((sample_weight > 0) | (estimator_weight < 0)) 

656 ) 

657 

658 return sample_weight, 1.0, estimator_error

664 estimator.fit(X, y, sample_weight=sample_weight) 

665 

666 y_predict = estimator.predict(X) 

667 

668 if iboost == 0: 

669 self.classes_ = getattr(estimator, "classes_", None) 

670 self.n_classes_ = len(self.classes_) 

671 

672 # Instances incorrectly classified 

673 incorrect = y_predict != y 

674 

675 # Error fraction 

676 estimator_error = np.mean(np.average(incorrect, weights=sample_weight, axis=0)) 

677 

678 # Stop if classification is perfect 

679 if estimator_error <= 0: 

680 return sample_weight, 1.0, 0.0 

681 

682 n_classes = self.n_classes_ 

683 

684 # Stop if the error is at least as bad as random guessing 

685 if estimator_error >= 1.0 - (1.0 / n_classes): 

686 self.estimators_.pop(-1) 

687 if len(self.estimators_) == 0: 

688 raise ValueError( 

689 "BaseClassifier in AdaBoostClassifier " 

690 "ensemble is worse than random, ensemble " 

691 "can not be fit." 

692 ) 

693 return None, None, None 

694 

695 # Boost weight using multi-class AdaBoost SAMME alg 

696 estimator_weight = self.learning_rate * ( 

697 np.log((1.0 - estimator_error) / estimator_error) + np.log(n_classes - 1.0) 

698 ) 

699 

700 # Only boost the weights if it will fit again 

701 if not iboost == self.n_estimators - 1: 

702 # Only boost positive weights 

703 sample_weight = np.exp( 

704 np.log(sample_weight) 

705 + estimator_weight * incorrect * (sample_weight > 0) 

706 ) 

707 

708 return sample_weight, estimator_weight, estimator_error

768 def decision_function(self, X): 

769 """Compute the decision function of ``X``. 

770 

771 Parameters 

772 ---------- 

773 X : {array-like, sparse matrix} of shape (n_samples, n_features) 

774 The training input samples. Sparse matrix can be CSC, CSR, COO, 

775 DOK, or LIL. COO, DOK, and LIL are converted to CSR. 

776 

777 Returns 

778 ------- 

779 score : ndarray of shape of (n_samples, k) 

780 The decision function of the input samples. The order of 

781 outputs is the same as that of the :term:`classes_` attribute. 

782 Binary classification is a special cases with ``k == 1``, 

783 otherwise ``k==n_classes``. For binary classification, 

784 values closer to -1 or 1 mean more like the first or second 

785 class in ``classes_``, respectively. 

786 """ 

787 check_is_fitted(self) 

788 X = self._check_X(X) 

789 

790 n_classes = self.n_classes_ 

791 classes = self.classes_[:, np.newaxis] 

792 

793 # TODO(1.6): Remove, because "algorithm" param will be deprecated in 1.6 

794 if self.algorithm == "SAMME.R": 

795 # The weights are all 1. for SAMME.R 

796 pred = sum( 

797 _samme_proba(estimator, n_classes, X) for estimator in self.estimators_ 

798 ) 

799 else: # self.algorithm == "SAMME" 

800 pred = sum( 

801 np.where( 

802 (estimator.predict(X) == classes).T, 

803 w, 

804 -1 / (n_classes - 1) * w, 

805 ) 

806 for estimator, w in zip(self.estimators_, self.estimator_weights_) 

807 ) 

808 

809 pred /= self.estimator_weights_.sum() 

810 if n_classes == 2: 

811 pred[:, 0] *= -1 

812 return pred.sum(axis=1) 

813 return pred

815 def staged_decision_function(self, X): 

816 """Compute decision function of ``X`` for each boosting iteration. 

817 

818 This method allows monitoring (i.e. determine error on testing set) 

819 after each boosting iteration. 

820 

821 Parameters 

822 ---------- 

823 X : {array-like, sparse matrix} of shape (n_samples, n_features) 

824 The training input samples. Sparse matrix can be CSC, CSR, COO, 

825 DOK, or LIL. COO, DOK, and LIL are converted to CSR. 

826 

827 Yields 

828 ------ 

829 score : generator of ndarray of shape (n_samples, k) 

830 The decision function of the input samples. The order of 

831 outputs is the same of that of the :term:`classes_` attribute. 

832 Binary classification is a special cases with ``k == 1``, 

833 otherwise ``k==n_classes``. For binary classification, 

834 values closer to -1 or 1 mean more like the first or second 

835 class in ``classes_``, respectively. 

836 """ 

837 check_is_fitted(self) 

838 X = self._check_X(X) 

839 

840 n_classes = self.n_classes_ 

841 classes = self.classes_[:, np.newaxis] 

842 pred = None 

843 norm = 0.0 

844 

845 for weight, estimator in zip(self.estimator_weights_, self.estimators_): 

846 norm += weight 

847 

848 # TODO(1.6): Remove, because "algorithm" param will be deprecated in 

849 # 1.6 

850 if self.algorithm == "SAMME.R": 

851 # The weights are all 1. for SAMME.R 

852 current_pred = _samme_proba(estimator, n_classes, X) 

853 else: # elif self.algorithm == "SAMME": 

854 current_pred = np.where( 

855 (estimator.predict(X) == classes).T, 

856 weight, 

857 -1 / (n_classes - 1) * weight, 

858 ) 

859 

860 if pred is None: 

861 pred = current_pred 

862 else: 

863 pred += current_pred 

864 

865 if n_classes == 2: 

866 tmp_pred = np.copy(pred) 

867 tmp_pred[:, 0] *= -1 

868 yield (tmp_pred / norm).sum(axis=1) 

869 else: 

870 yield pred / norm

872 @staticmethod 

873 def _compute_proba_from_decision(decision, n_classes): 

874 """Compute probabilities from the decision function. 

875 

876 This is based eq. (15) of [1] where: 

877 p(y=c|X) = exp((1 / K-1) f_c(X)) / sum_k(exp((1 / K-1) f_k(X))) 

878 = softmax((1 / K-1) * f(X)) 

879 

880 References 

881 ---------- 

882 .. [1] J. Zhu, H. Zou, S. Rosset, T. Hastie, "Multi-class AdaBoost", 

883 2009. 

884 """ 

885 if n_classes == 2: 

886 decision = np.vstack([-decision, decision]).T / 2 

887 else: 

888 decision /= n_classes - 1 

889 return softmax(decision, copy=False)

971class AdaBoostRegressor(_RoutingNotSupportedMixin, RegressorMixin, BaseWeightBoosting): 

972 """An AdaBoost regressor. 

973 

974 An AdaBoost [1] regressor is a meta-estimator that begins by fitting a 

975 regressor on the original dataset and then fits additional copies of the 

976 regressor on the same dataset but where the weights of instances are 

977 adjusted according to the error of the current prediction. As such, 

978 subsequent regressors focus more on difficult cases. 

979 

980 This class implements the algorithm known as AdaBoost.R2 [2]. 

981 

982 Read more in the :ref:`User Guide <adaboost>`. 

983 

984 .. versionadded:: 0.14 

985 

986 Parameters 

987 ---------- 

988 estimator : object, default=None 

989 The base estimator from which the boosted ensemble is built. 

990 If ``None``, then the base estimator is 

991 :class:`~sklearn.tree.DecisionTreeRegressor` initialized with 

992 `max_depth=3`. 

993 

994 .. versionadded:: 1.2 

995 `base_estimator` was renamed to `estimator`. 

996

999 In case of perfect fit, the learning procedure is stopped early. 

1000 Values must be in the range `[1, inf)`. 

1001

1003 Weight applied to each regressor at each boosting iteration. A higher 

1004 learning rate increases the contribution of each regressor. There is 

1005 a trade-off between the `learning_rate` and `n_estimators` parameters. 

1006 Values must be in the range `(0.0, inf)`. 

1007 

1008 loss : {'linear', 'square', 'exponential'}, default='linear' 

1009 The loss function to use when updating the weights after each 

1010 boosting iteration. 

1011 

1012 random_state : int, RandomState instance or None, default=None 

1013 Controls the random seed given at each `estimator` at each 

1014 boosting iteration. 

1015 Thus, it is only used when `estimator` exposes a `random_state`. 

1016 In addition, it controls the bootstrap of the weights used to train the 

1017 `estimator` at each boosting iteration. 

1018 Pass an int for reproducible output across multiple function calls. 

1019 See :term:`Glossary <random_state>`. 

1020

1026 .. versionadded:: 1.2 

1027 `base_estimator_` was renamed to `estimator_`.

1036 Regression error for each estimator in the boosted ensemble.

1039 The impurity-based feature importances if supported by the 

1040 ``estimator`` (when based on decision trees). 

1041

1043 high cardinality features (many unique values). See 

1044 :func:`sklearn.inspection.permutation_importance` as an alternative.

1046 n_features_in_ : int 

1047 Number of features seen during :term:`fit`. 

1048 

1049 .. versionadded:: 0.24 

1050 

1051 feature_names_in_ : ndarray of shape (`n_features_in_`,) 

1052 Names of features seen during :term:`fit`. Defined only when `X` 

1053 has feature names that are all strings.

1055 .. versionadded:: 1.0

1059 AdaBoostClassifier : An AdaBoost classifier. 

1060 GradientBoostingRegressor : Gradient Boosting Classification Tree. 

1061 sklearn.tree.DecisionTreeRegressor : A decision tree regressor.

1063 References 

1064 ---------- 

1065 .. [1] Y. Freund, R. Schapire, "A Decision-Theoretic Generalization of 

1066 on-Line Learning and an Application to Boosting", 1995. 

1067 

1068 .. [2] H. Drucker, "Improving Regressors using Boosting Techniques", 1997.

1071 --------

1117 def _boost(self, iboost, X, y, sample_weight, random_state): 

1118 """Implement a single boost for regression 

1119 

1120 Perform a single boost according to the AdaBoost.R2 algorithm and 

1121 return the updated sample weights. 

1122 

1123 Parameters 

1124 ---------- 

1125 iboost : int 

1126 The index of the current boost iteration. 

1127 

1128 X : {array-like, sparse matrix} of shape (n_samples, n_features) 

1129 The training input samples. 

1130 

1131 y : array-like of shape (n_samples,) 

1132 The target values (class labels in classification, real numbers in 

1133 regression). 

1134 

1135 sample_weight : array-like of shape (n_samples,) 

1136 The current sample weights. 

1137 

1138 random_state : RandomState 

1139 The RandomState instance used if the base estimator accepts a 

1140 `random_state` attribute. 

1141 Controls also the bootstrap of the weights used to train the weak 

1142 learner. 

1143 replacement. 

1144 

1145 Returns 

1146 ------- 

1147 sample_weight : array-like of shape (n_samples,) or None 

1148 The reweighted sample weights. 

1149 If None then boosting has terminated early. 

1150 

1151 estimator_weight : float 

1152 The weight for the current boost. 

1153 If None then boosting has terminated early. 

1154 

1155 estimator_error : float 

1156 The regression error for the current boost. 

1157 If None then boosting has terminated early. 

1158 """ 

1159 estimator = self._make_estimator(random_state=random_state) 

1160 

1161 # Weighted sampling of the training set with replacement 

1162 bootstrap_idx = random_state.choice( 

1163 np.arange(_num_samples(X)), 

1164 size=_num_samples(X), 

1165 replace=True, 

1166 p=sample_weight, 

1167 ) 

1168 

1169 # Fit on the bootstrapped sample and obtain a prediction 

1170 # for all samples in the training set 

1171 X_ = _safe_indexing(X, bootstrap_idx) 

1172 y_ = _safe_indexing(y, bootstrap_idx) 

1173 estimator.fit(X_, y_) 

1174 y_predict = estimator.predict(X) 

1175 

1176 error_vect = np.abs(y_predict - y) 

1177 sample_mask = sample_weight > 0 

1178 masked_sample_weight = sample_weight[sample_mask] 

1179 masked_error_vector = error_vect[sample_mask] 

1180 

1181 error_max = masked_error_vector.max() 

1182 if error_max != 0: 

1183 masked_error_vector /= error_max 

1184 

1185 if self.loss == "square": 

1186 masked_error_vector **= 2 

1187 elif self.loss == "exponential": 

1188 masked_error_vector = 1.0 - np.exp(-masked_error_vector) 

1189 

1190 # Calculate the average loss 

1191 estimator_error = (masked_sample_weight * masked_error_vector).sum() 

1192 

1193 if estimator_error <= 0: 

1194 # Stop if fit is perfect 

1195 return sample_weight, 1.0, 0.0 

1196 

1197 elif estimator_error >= 0.5: 

1198 # Discard current estimator only if it isn't the only one 

1199 if len(self.estimators_) > 1: 

1200 self.estimators_.pop(-1) 

1201 return None, None, None 

1202 

1203 beta = estimator_error / (1.0 - estimator_error) 

1204 

1205 # Boost weight using AdaBoost.R2 alg 

1206 estimator_weight = self.learning_rate * np.log(1.0 / beta) 

1207 

1208 if not iboost == self.n_estimators - 1: 

1209 sample_weight[sample_mask] *= np.power( 

1210 beta, (1.0 - masked_error_vector) * self.learning_rate 

1211 ) 

1212 

1213 return sample_weight, estimator_weight, estimator_error

1215 def _get_median_predict(self, X, limit): 

1216 # Evaluate predictions of all estimators 

1217 predictions = np.array([est.predict(X) for est in self.estimators_[:limit]]).T 

1218 

1219 # Sort the predictions 

1220 sorted_idx = np.argsort(predictions, axis=1) 

1221 

1222 # Find index of median prediction for each sample 

1223 weight_cdf = stable_cumsum(self.estimator_weights_[sorted_idx], axis=1) 

1224 median_or_above = weight_cdf >= 0.5 * weight_cdf[:, -1][:, np.newaxis] 

1225 median_idx = median_or_above.argmax(axis=1) 

1226 

1227 median_estimators = sorted_idx[np.arange(_num_samples(X)), median_idx] 

1228 

1229 # Return median predictions 

1230 return predictions[np.arange(_num_samples(X)), median_estimators]
