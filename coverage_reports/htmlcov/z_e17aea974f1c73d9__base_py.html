20class SelectorMixin(TransformerMixin, metaclass=ABCMeta): 

21 """ 

22 Transformer mixin that performs feature selection given a support mask 

23 

24 This mixin provides a feature selector implementation with `transform` and 

25 `inverse_transform` functionality given an implementation of 

26 `_get_support_mask`. 

27 

28 Examples 

29 -------- 

30 >>> import numpy as np 

31 >>> from sklearn.datasets import load_iris 

32 >>> from sklearn.base import BaseEstimator 

33 >>> from sklearn.feature_selection import SelectorMixin 

34 >>> class FeatureSelector(SelectorMixin, BaseEstimator): 

35 ... def fit(self, X, y=None): 

36 ... self.n_features_in_ = X.shape[1] 

37 ... return self 

38 ... def _get_support_mask(self): 

39 ... mask = np.zeros(self.n_features_in_, dtype=bool) 

40 ... mask[:2] = True # select the first two features 

41 ... return mask 

42 >>> X, y = load_iris(return_X_y=True) 

43 >>> FeatureSelector().fit_transform(X, y).shape 

44 (150, 2) 

45 """

47 def get_support(self, indices=False): 

48 """ 

49 Get a mask, or integer index, of the features selected. 

50 

51 Parameters 

52 ---------- 

53 indices : bool, default=False 

54 If True, the return value will be an array of integers, rather 

55 than a boolean mask. 

56 

57 Returns 

58 ------- 

59 support : array 

60 An index that selects the retained features from a feature vector. 

61 If `indices` is False, this is a boolean array of shape 

62 [# input features], in which an element is True iff its 

63 corresponding feature is selected for retention. If `indices` is 

64 True, this is an integer array of shape [# output features] whose 

65 values are indices into the input feature vector. 

66 """ 

67 mask = self._get_support_mask() 

68 return mask if not indices else np.where(mask)[0]
